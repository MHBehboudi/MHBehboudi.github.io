about

Ph.D. candidate in Cognition & Neuroscience at UT Dallas with an Electrical Engineering foundation. I specialize in the full computational stack for physiological signals: universal data readers across formats (EDF, BDF, HDF5, BrainVision), automated preprocessing handling cross-device variability, and foundation model architectures that learn transferable representations from unlabeled data for any downstream task—BCI, clinical diagnostics, or wearable health.

I've built a novel connectivity algorithm (MDPC) adopted by multiple research groups, engineered automated pipelines processing 500+ GB across heterogeneous recording systems, and designed a clinical diagnostic system achieving 95% accuracy from single-channel recordings.

class EEGFoundationDataset(Dataset): """Serves windowed biosignal patches from heterogeneous HDF5 files. Handles variable channels, sampling rates, devices.""" def __init__(self, h5_files, window_sec=10.0, overlap=0.5, target_sfreq=100.0): self.window_samples = int(window_sec * target_sfreq) self.step = int(self.window_samples * (1 - overlap)) self.index = [] for path in h5_files: with h5py.File(path, 'r') as f: grp = f[sub_key][ses_key] n_chan, n_samp = grp['data'].shape for start in range(0, n_samp - self.window_samples, self.step): self.index.append({ 'filepath': path, 'start': start, 'n_channels': n_chan }) def __getitem__(self, idx): info = self.index[idx] # Open HDF5 lazily — fork-safe for DataLoader workers with h5py.File(info['filepath'], 'r') as f: data = f[grp]['data'][:, start:end] return {'data': torch.from_numpy(data), 'n_channels': info['n_channels']}

collate.py — Variable-Channel Attention MaskPyTorch
def eeg_collate_fn(batch): """Pad to max channels in batch, create attention mask. Enables Transformer to handle mixed-montage data.""" max_ch = max(b['n_channels'] for b in batch) B = len(batch) padded = torch.zeros(B, max_ch, window_samples) mask = torch.zeros(B, max_ch) for i, item in enumerate(batch): n = item['n_channels'] padded[i, :n] = item['data'] mask[i, :n] = 1.0 # real channels return {'data': padded, 'channel_mask': mask}

tokenizer.py — VQ-VAE Neural TokenizerPyTorch
class NeuralTokenizer(nn.Module): """VQ-VAE tokenizer: learns discrete codebook from spectral representations of biosignal patches. Target = Fourier spectrum (not raw signal).""" def __init__(self, codebook_size=8192, codebook_dim=64): self.encoder = TemporalConvEncoder(d_model=256) self.codebook = nn.Embedding(codebook_size, codebook_dim) self.decoder = TransformerDecoder(n_layers=4) self.amp_head = nn.Linear(codebook_dim, fft_bins) self.phs_head = nn.Linear(codebook_dim, fft_bins) def quantize(self, z): # L2-normalized cosine similarity (BEiT v2) z_norm = F.normalize(z, dim=-1) cb_norm = F.normalize(self.codebook.weight, dim=-1) indices = torch.argmin( torch.cdist(z_norm, cb_norm), dim=-1) return indices # straight-through estimator
